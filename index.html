window.ativar = async function() { try { // 🔥 pega status real do navegador const status = Notification.permission; if (status === "granted") { await OneSignal.User.PushSubscription.optIn(); const id = await OneSignal.User.getOnesignalId(); console.log("ID:", id); alert("✅ Notificações já estão ativas!"); return; } if (status === "denied") { alert("⚠️ Você bloqueou notificações.\nClique no cadeado 🔒 e permita."); return; } // 🔔 aqui sim pede permissão (funciona normal) const res = await Notification.requestPermission(); if (res === "granted") { await OneSignal.User.PushSubscription.optIn(); const id = await OneSignal.User.getOnesignalId(); console.log("ID:", id); alert("✅ Inscrito com sucesso!"); } } catch (e) { console.error(e); alert("Erro"); } };
